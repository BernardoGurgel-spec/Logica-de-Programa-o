algoritmo "CalculadoraBasica"

var
   n1, n2, resultado: real
   opcao: inteiro
   continuar: logico

inicio

   continuar <- verdadeiro

   enquanto (continuar = verdadeiro) faca

      limpatela

      escreval("===== CALCULADORA =====")
      escreval("1 - Soma")
      escreval("2 - Subtração")
      escreval("3 - Multiplicação")
      escreval("4 - Divisão")
      escreval("0 - Sair")
      escreva("Escolha uma opção: ")
      leia(opcao)

      se (opcao = 0) entao
         limpatela
         escreval("Saindo...")
         continuar <- falso
      senao

         limpatela
         escreval("Digite o primeiro número: ")
         leia(n1)
         limpatela
         escreval("Digite o segundo número: ")
         leia(n2)

         escolha(opcao)

            caso 1
               resultado <- n1 + n2
               limpatela
               escreval("Resultado: ", resultado)

            caso 2
               resultado <- n1 - n2
               limpatela
               escreval("Resultado: ", resultado)

            caso 3
               resultado <- n1 * n2
               limpatela
               escreval("Resultado: ", resultado)

            caso 4
               limpatela
               se (n2 = 0) entao
                  escreval("Erro: divisão por zero não é permitida!")
               senao
                  resultado <- n1 / n2
                  escreval("Resultado: ", resultado)
               fimse

            outrocaso
               limpatela
               escreval("Opção inválida!")

         fimescolha

         escreval("")
         escreval("Pressione ENTER para continuar...")
         leia(n1)

      fimse

   fimenquanto

fimalgoritmo
